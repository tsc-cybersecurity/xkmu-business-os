**Standard migration pattern** (applies to all 12 non-database routes):
```typescript
// BEFORE:
export async function POST(request: NextRequest, { params }: { params: Params }) {
  const auth = await getAuthContext(request)
  if (!auth) return apiUnauthorized()
  // ... handler body using auth.tenantId, auth.userId
}

// AFTER:
export async function POST(request: NextRequest, { params }: { params: Params }) {
  return withPermission(request, '<module>', '<action>', async (auth) => {
    // ... same handler body, unchanged
    // auth.tenantId, auth.userId, auth.role, auth.roleId all available
  })
}
```

**Special case — export/database/route.ts (streaming GET):**

```typescript
export async function GET(request: NextRequest) {

    const stream = new ReadableStream({ ... })  // same as before, unchanged
    return new NextResponse(stream, {
      status: 200,
      headers: {

  })
}
```
Do NOT wrap the stream in `apiSuccess()`. Return `new NextResponse(stream, ...)` directly.
