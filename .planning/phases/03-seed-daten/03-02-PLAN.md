**dauer -> estimatedDurationMinutes Parsing:**
```typescript
function parseDauer(dauer: string | undefined): number | null {
  if (!dauer) return null
  // "30 Min" -> 30, "60 Min/Monat" -> 60, "4-8 Std" -> 360 (Mittelpunkt * 60)
  // "5 Min/Tag" -> 5, "2 Std/Jahr" -> 120, "Einmalig" -> null, "Variabel" -> null
  const minMatch = dauer.match(/^(\d+)\s*Min/)
  if (minMatch) return parseInt(minMatch[1])
  const stdMatch = dauer.match(/^(\d+)(?:-(\d+))?\s*Std/)
  if (stdMatch) {
    const low = parseInt(stdMatch[1])
    const high = stdMatch[2] ? parseInt(stdMatch[2]) : low
    return Math.round(((low + high) / 2) * 60)
  }
  return null
}
```

```typescript
const MOD = 'SopSeed'

export async function seedSopCatalog(tenantId: string) {
  logger.info('Seeding SOP Catalog...', { module: MOD })

  // 1. Deliverable-Lookup-Map aufbauen
  const allDeliverables = await db.select({ id: deliverables.id, name: deliverables.name })
    .from(deliverables)
    .where(eq(deliverables.tenantId, tenantId))
  const deliverableByName = new Map(allDeliverables.map(d => [d.name, d.id]))
  logger.info(`${allDeliverables.length} Deliverables geladen fuer Lookup`, { module: MOD })

  let created = 0, skipped = 0

  for (const sop of SOPS) {
    // 2. Idempotenz pruefen (zweistufig)
    if (sop.source_task_id) {
      const [existing] = await db.select({ id: sopDocuments.id }).from(sopDocuments)
        .where(and(
          eq(sopDocuments.tenantId, tenantId),
          eq(sopDocuments.sourceTaskId, sop.source_task_id),
          isNull(sopDocuments.deletedAt),
        ))
      if (existing) { skipped++; continue }
    } else {
      const [existing] = await db.select({ id: sopDocuments.id }).from(sopDocuments)
        .where(and(
          eq(sopDocuments.tenantId, tenantId),
          eq(sopDocuments.title, sop.title),
          isNull(sopDocuments.deletedAt),
        ))
      if (existing) { skipped++; continue }
    }

    // 3. Deliverable-Verknuepfung aufloesen
    let producesDeliverableId: string | null = null
    if (sop.produces_deliverable) {
      const delivName = DELIVERABLE_CODE_TO_NAME[sop.produces_deliverable] ?? null
      if (delivName) {
        producesDeliverableId = deliverableByName.get(delivName) ?? null
        if (!producesDeliverableId) {
          logger.warn(`Deliverable nicht gefunden: ${sop.produces_deliverable} -> ${delivName}`, { module: MOD })
        }
      }
    }

    // 4. SOP direkt einfuegen (db.insert, NICHT SopService.create)
    await db.insert(sopDocuments).values({
      tenantId,
      title: sop.title,
      category: sop.category,
      status: sop.status ?? 'draft',
      version: '1.0.0',
      automationLevel: sop.automation_level ?? null,
      aiCapable: sop.automation_level !== 'manual',
      maturityLevel: sop.maturity_level ?? null,
      estimatedDurationMinutes: parseDauer(sop.dauer),
      subprocess: sop.subprocess ?? null,
      sourceTaskId: sop.source_task_id ?? null,
      producesDeliverableId,
      purpose: sop.output_beschreibung ?? null,
      scope: null,
      tools: [],
      tags: [],
    })
    created++
  }

  logger.info(`SOP Catalog seeded: ${created} erstellt, ${skipped} bereits vorhanden`, { module: MOD })
}
```
