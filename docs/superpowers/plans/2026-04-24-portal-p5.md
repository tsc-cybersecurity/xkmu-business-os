  companyId: uuid('company_id').notNull().references(() => companies.id, { onDelete: 'cascade' }),
  senderId: uuid('sender_id').references(() => users.id, { onDelete: 'set null' }),

  bodyText: text('body_text').notNull(),

  readByAdminAt: timestamp('read_by_admin_at', { withTimezone: true }),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
}, (table) => [
  index('idx_chat_messages_company_created').on(table.companyId, table.createdAt),
  index('idx_chat_messages_sender').on(table.senderId),
])

export type ChatMessage = typeof chatMessages.$inferSelect
export type NewChatMessage = typeof chatMessages.$inferInsert

Migration 012:

```sql
-- ============================================================
-- Migration 012: Portal P5 — chat_messages
-- Idempotent.
-- ============================================================

CREATE TABLE IF NOT EXISTS chat_messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  sender_id UUID REFERENCES users(id) ON DELETE SET NULL,
  sender_role VARCHAR(50) NOT NULL,
  body_text TEXT NOT NULL,
  read_by_portal_at TIMESTAMPTZ,
  read_by_admin_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_chat_messages_company_created ON chat_messages(company_id, created_at DESC);
CREATE INDEX IF NOT EXISTS idx_chat_messages_sender ON chat_messages(sender_id);
```

### GET /messages
`withPortalAuth` → `ChatService.listForCompany(auth.companyId, sinceParam, 100)`. Optional `?since=<iso>`.

### POST /messages
zod: `{ bodyText: z.string().min(1).max(5000) }`.
Rate-Limit: 30 msg/min per IP via existing `rateLimit(request, 'portal-chat', 30, 60_000)`.
Server setzt `senderId=auth.userId`, `senderRole='portal_user'`, `companyId=auth.companyId`.
Audit: `portal.chat_message_sent`, entityType='chat_message', entityId=message.id, payload `{ companyId, messageId, characterCount }`. Fail-safe.

### PATCH /mark-read
`ChatService.markReadByPortal(auth.companyId)`. No audit (high-frequency, low value).

### GET /unread-count
`ChatService.unreadCountForPortal(auth.companyId)` → `{ unread: number }`.

### /portal/chat

Client component. Polling via `setInterval(15000)`. Pausiert bei `document.hidden` (Page Visibility API).

```tsx
'use client'
import { useEffect, useRef, useState } from 'react'
...

export default function PortalChatPage() {
  const [messages, setMessages] = useState<ChatMessage[]>([])
  const [draft, setDraft] = useState('')
  const [sending, setSending] = useState(false)
  const bottomRef = useRef<HTMLDivElement>(null)
  const lastCreatedAt = useRef<string | null>(null)

  const load = async (initial = false) => {
    const url = initial
      ? '/api/v1/portal/me/chat/messages'
      : `/api/v1/portal/me/chat/messages?since=${encodeURIComponent(lastCreatedAt.current ?? '')}`
    const res = await fetch(url)
    const data = await res.json()
    if (data?.success) {
      if (initial) setMessages(data.data)
      else setMessages(prev => [...prev, ...data.data])
      if (data.data.length > 0) lastCreatedAt.current = data.data[data.data.length - 1].createdAt
      // Scroll to bottom
      requestAnimationFrame(() => bottomRef.current?.scrollIntoView({ behavior: 'smooth' }))
      // Mark as read
      await fetch('/api/v1/portal/me/chat/mark-read', { method: 'PATCH' })
    }
  }

  useEffect(() => {
    load(true)
    const interval = setInterval(() => {
      if (!document.hidden) load(false)
    }, 15000)
    return () => clearInterval(interval)
  }, [])

  const send = async () => {
    if (!draft.trim()) return
    setSending(true)
    try {
      const res = await fetch('/api/v1/portal/me/chat/messages', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ bodyText: draft.trim() }),
      })
      const data = await res.json()
      if (data?.success) {
        setDraft('')
        await load(false)  // reload delta to pick up own message
      } else {
        toast.error(data.error?.message || 'Senden fehlgeschlagen')
      }
    } finally {
      setSending(false)
    }
  }

  // Render: messages list, bottom textarea, send button, Enter to send
}
```

Message-Render: Eigene Messages rechts (blau), Admin-Messages links (grau), Time + Read-Receipt unter der eigenen Message (✓ einfach gesendet, ✓✓ vom Admin gelesen).
